ConsumableItem ScriptableObject実装

using UnityEngine;
using Game.Items;

// 消費アイテム用ScriptableObject
[CreateAssetMenu(fileName = "NewConsumable", menuName = "Game/Consumable")]
public class ConsumableItem : ScriptableObject, IPlayerItem
{
    [Header("基本情報")]
    public string itemID;
    public string itemName;
    public Sprite icon;
    [TextArea] public string description;
    public int maxStack = 500; // スタック上限

    // IPlayerItem インターフェース実装
    public string ItemID => itemID;
    public string ItemName => itemName;
    public Sprite Icon => icon;
    public ItemType Type => ItemType.Consumable;
    public string Description => description;
}

Inventory System with Generic Design

// IPlayerItemインターフェースを使用したインベントリシステム
public class InventorySystem : MonoBehaviour
{
    public static System.Action<IPlayerItem, int> OnItemAdded;
    public static System.Action<IPlayerItem, int> OnItemRemoved;

    [SerializeField] private List<InventorySlot> slots;
    [SerializeField] private int maxSlots = 30;

    public bool AddItem(IPlayerItem item, int quantity = 1)
    {
        // ConsumableItemかどうかで処理を分岐
        if (item is ConsumableItem consumable)
        {
            // スタック可能なアイテムの処理
            var existingSlot = slots.Find(slot => 
                slot.item.ItemID == item.ItemID && 
                slot.quantity < consumable.maxStack);

            if (existingSlot != null)
            {
                int addAmount = Mathf.Min(quantity, 
                    consumable.maxStack - existingSlot.quantity);
                existingSlot.quantity += addAmount;
                OnItemAdded?.Invoke(item, addAmount);
                return true;
            }
        }

        // 新スロットに追加
        if (slots.Count < maxSlots)
        {
            slots.Add(new InventorySlot(item, quantity));
            OnItemAdded?.Invoke(item, quantity);
            return true;
        }
        return false;
    }
}
