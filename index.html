ItemData ScriptableObject Implementation

// アイテムデータの基本構造を定義するScriptableObject
[CreateAssetMenu(fileName = "New Item", menuName = "RPG/Item Data")]
public class ItemData : ScriptableObject
{
    [Header("Basic Info")]
    public string itemName;
    [TextArea(3, 5)]
    public string description;
    public Sprite icon;
    public ItemType itemType;
    public int maxStackSize = 1;

    [Header("Stats Modifiers")]
    public StatModifier[] statModifiers;

    // アイテム使用時の効果を定義
    public virtual void UseItem(PlayerController player)
    {
        foreach (var modifier in statModifiers)
        {
            player.stats.ApplyModifier(modifier);
        }
    }
}

Inventory System with Event Architecture

// Observer Patternを使用したインベントリシステム
public class InventorySystem : MonoBehaviour
{
    public static System.Action<ItemData, int> OnItemAdded;
    public static System.Action<ItemData, int> OnItemRemoved;

    [SerializeField] private List<InventorySlot> inventory;
    [SerializeField] private int maxSlots = 30;

    public bool AddItem(ItemData itemData, int quantity = 1)
    {
        // スタック可能なアイテムの処理
        var existingSlot = inventory.Find(slot => 
            slot.itemData == itemData && 
            slot.quantity < itemData.maxStackSize);

        if (existingSlot != null)
        {
            int addAmount = Mathf.Min(quantity, 
                itemData.maxStackSize - existingSlot.quantity);
            existingSlot.quantity += addAmount;
            OnItemAdded?.Invoke(itemData, addAmount);
            return true;
        }

        // 新しいスロットに追加
        if (inventory.Count < maxSlots)
        {
            inventory.Add(new InventorySlot(itemData, quantity));
            OnItemAdded?.Invoke(itemData, quantity);
            return true;
        }

        return false; // インベントリが満杯
    }
}

UI System Integration

// UIManagerがInventorySystemのイベントを購読
public class InventoryUIManager : MonoBehaviour
{
    [SerializeField] private Transform slotParent;
    [SerializeField] private InventorySlotUI slotPrefab;
    
    private List<InventorySlotUI> slotUIs = new List<InventorySlotUI>();

    private void OnEnable()
    {
        // イベントの購読
        InventorySystem.OnItemAdded += UpdateUI;
        InventorySystem.OnItemRemoved += UpdateUI;
    }

    private void OnDisable()
    {
        // メモリリーク防止
        InventorySystem.OnItemAdded -= UpdateUI;
        InventorySystem.OnItemRemoved -= UpdateUI;
    }

    private void UpdateUI(ItemData item, int quantity)
    {
        // UI更新ロジック（データバインディング）
        var inventory = InventorySystem.Instance.GetInventory();
        
        for (int i = 0; i < slotUIs.Count; i++)
        {
            if (i < inventory.Count)
                slotUIs[i].UpdateSlot(inventory[i]);
            else
                slotUIs[i].ClearSlot();
        }
    }
}
